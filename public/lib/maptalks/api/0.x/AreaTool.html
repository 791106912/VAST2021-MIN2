AreaTool

A map tool to help measure area on the map

new AreaTool(optionsopt) [source] 

var areaTool = new AreaTool({
    'once' : true,
    'symbol': {
      'lineColor' : '#34495e',
      'lineWidth' : 2
    },
    'vertexSymbol' : {
      'markerType'        : 'ellipse',
      'markerFill'        : '#1bbc9b',
      'markerLineColor'   : '#000',
      'markerLineWidth'   : 3,
      'markerWidth'       : 10,
     'markerHeight'      : 10
   },
   'language' : 'en-US'
 }).addTo(map);
